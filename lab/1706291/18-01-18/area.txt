{
	float r,br,lt,bt,AC,AR,AT;
	int lr;
	cout<<"Enter The radius Of The Circle\n";

	cout<<"Enter The Lenght And Breath Of Rectangle\n";

float area(float rc)
{
	return(3.1415926*rc*rc);
}
float area(int l,float b)
{
	return(l*b);
}
float area(float len,float bre)
{
	return(0.5*len*bre);
}
